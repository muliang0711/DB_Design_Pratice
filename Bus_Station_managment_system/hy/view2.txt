CREATE OR REPLACE VIEW v_ticket_overview AS
SELECT 
    t.ticketId,
    t.busScheduleId,
    c.firstName || ' ' || c.lastName AS customerName,
    t.seatNo,
    t.price,
    t.type,
    t.status,
    t.createdAt
FROM ticket t
LEFT JOIN customer c 
    ON t.customerId = c.customerId;

CREATE INDEX IDX_TICKET_STATUS
  ON ticket(status);

CREATE INDEX IDX_TICKET_PRICE        ON ticket(price);

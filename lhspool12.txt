SQL> CREATE OR REPLACE VIEW BusMaintenanceStatsView AS
  2  SELECT
  3      -- Makes report prettier by printing the company name
  4      -- only for the first row in each group.
  5      -- CASE
  6      --     WHEN ROW_NUMBER() OVER (PARTITION BY c.companyName ORDER BY b.busID) = 1
  7      --     THEN c.companyName
  8      --     ELSE NULL
  9      -- END AS companyName,
 10      c.companyName       AS companyName,
 11      b.busID             AS busID,
 12      b.plateNo           AS plateNo,
 13      AVG(interval_days)  AS avgIntervalDays
 14  FROM (
 15      SELECT
 16          bm.busID,
 17          ROUND(
 18              CAST(LEAD(bm.maintenanceDoneDate) OVER (PARTITION BY bm.busID ORDER BY bm.maintenanceDoneDate) AS DATE)
 19              - CAST(bm.maintenanceDoneDate AS DATE)
 20          ) AS interval_days
 21      FROM BusMaintenance bm
 22      JOIN MaintenanceService ms
 23        ON bm.serviceID = ms.serviceID
 24      WHERE UPPER(ms.category) IN ('MODERATE', 'MAJOR')
 25  ) intervals
 26  JOIN Bus b ON intervals.busID = b.busID
 27  JOIN BusCompany c ON b.companyID = c.companyID
 28  WHERE interval_days IS NOT NULL
 29  GROUP BY b.busID, c.companyName, b.plateNo
 30  ORDER BY c.companyName, b.busID;

SQL> CREATE OR REPLACE VIEW BusMaintenanceStatsView AS
  2  SELECT
  3      -- Makes report prettier by printing the company name
  4      -- only for the first row in each group.
  5      -- CASE
  6      --     WHEN ROW_NUMBER() OVER (PARTITION BY c.companyName ORDER BY b.busID) = 1
  7      --     THEN c.companyName
  8      --     ELSE NULL
  9      -- END AS companyName,
 10      c.companyName       AS companyName,
 11      b.busID             AS busID,
 12      b.plateNo           AS plateNo,
 13      AVG(interval_days)  AS avgIntervalDays
 14  FROM (
 15      SELECT
 16          bm.busID,
 17          ROUND(
 18              CAST(LEAD(bm.maintenanceDoneDate) OVER (PARTITION BY bm.busID ORDER BY bm.maintenanceDoneDate) AS DATE)
 19              - CAST(bm.maintenanceDoneDate AS DATE)
 20          ) AS interval_days
 21      FROM BusMaintenance bm
 22      JOIN MaintenanceService ms
 23        ON bm.serviceID = ms.serviceID
 24      WHERE UPPER(ms.category) IN ('MODERATE', 'MAJOR')
 25  ) intervals
 26  JOIN Bus b ON intervals.busID = b.busID
 27  JOIN BusCompany c ON b.companyID = c.companyID
 28  WHERE interval_days IS NOT NULL
 29  GROUP BY b.busID, c.companyName, b.plateNo
 30  ORDER BY c.companyName, b.busID;

SQL> CREATE OR REPLACE VIEW BusMaintenanceStatsView AS
  2  SELECT
  3      -- Makes report prettier by printing the company name
  4      -- only for the first row in each group.
  5      -- CASE
  6      --     WHEN ROW_NUMBER() OVER (PARTITION BY c.companyName ORDER BY b.busID) = 1
  7      --     THEN c.companyName
  8      --     ELSE NULL
  9      -- END AS companyName,
 10      c.companyName       AS companyName,
 11      b.busID             AS busID,
 12      b.plateNo           AS plateNo,
 13      ROUND(AVG(interval_days))  AS avgIntervalDays
 14  FROM (
 15      SELECT
 16          bm.busID,
 17          CAST(LEAD(bm.maintenanceDoneDate) OVER (PARTITION BY bm.busID ORDER BY bm.maintenanceDoneDate) AS DATE)
 18          - CAST(bm.maintenanceDoneDate AS DATE)
 19              AS interval_days
 20      FROM BusMaintenance bm
 21      JOIN MaintenanceService ms
 22        ON bm.serviceID = ms.serviceID
 23      WHERE UPPER(ms.category) IN ('MODERATE', 'MAJOR')
 24  ) intervals
 25  JOIN Bus b ON intervals.busID = b.busID
 26  JOIN BusCompany c ON b.companyID = c.companyID
 27  WHERE interval_days IS NOT NULL
 28  GROUP BY b.busID, c.companyName, b.plateNo
 29  ORDER BY c.companyName, b.busID;
